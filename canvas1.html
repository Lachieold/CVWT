        📅 <strong>Today’s Date:</strong> <span id="date"></span><br>

        🗓️ <strong>Current Week:</strong> <span id="week"></span>

    </div>


    <script>

        function getCurrentWeek() {

            const today = new Date();

            const week0Start = new Date("2024-02-04"); // Week 0 starts today

            const week1Start = new Date("2025-03-01"); // Week 1 starts March 1, 2025


            let currentWeek = "Week 0"; // Default to Week 0

            if (today >= week1Start) {

                const weekNumber = Math.floor((today - week1Start) / (7 * 24 * 60 * 60 * 1000)) + 1;

                currentWeek = weekNumber >= 8 ? "Week 8" : `Week ${weekNumber}`;

            }


            document.getElementById("date").textContent = today.toDateString();

            document.getElementById("week").textContent = currentWeek;

        }


        getCurrentWeek();

    </script>
